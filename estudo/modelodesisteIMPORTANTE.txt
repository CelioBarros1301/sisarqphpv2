	<script>
	$(document).ready(function(){
	    $('.form-modal-ajax').on('click',function(){

			$('select, input, textarea').removeClass('is-invalid');
			
	        var url = $(this).attr('data-href');
	        var modalId = $(this).attr('data-target');
	        $(modalId + ' form').attr('action', url);

			$('.delete').addClass('invisible');
    		$('.loader').removeClass('invisible');

			$(modalId + ' form').find("input[type=text], textarea, select").val('');
			$(modalId + ' form').find("input[type=checkbox]").prop('checked', false).change();			

			$.getJSON( url, function( data ) {

				$.each(data, function(index, element) {

					if($( "*[name='"+index+"']" ).attr('type') == "checkbox")
					{
						if(element)
						{
							$( "*[name='"+index+"']" ).prop('checked', true).change();
						} 

						return true;
					} 

					$( "*[name='"+index+"']" ).val(element);
				});

				if(data.length != 0)
					$('.delete').removeClass('invisible');	
				
				$('.loader').addClass('invisible');
				
				$(modalId).modal({show:true});

			});
	        
	    }); 


	    $('.form-ajax').submit(function(e){
	
			$post = $.post($(this).attr('action'), $(this).serialize());

			$('select, input, textarea').removeClass('is-invalid');

			$post.done(function( data ) {
	         	if(data.redirect) {
	         		window.location.href = data.redirect;
	         	}

	         	if(data.errors) {
	         		$.each(data.errors, function( index, value ) {
	         			$('#'+index+'-error').text(value);
	         			$('#'+index).addClass('is-invalid');
					});
	         	}

	         	if(data.message) {
	         		$('#message').text(data.message);
					$('#message').show();
	         	}
			});

	    	e.preventDefault();
	    });


	});
	</script>
